# Function to read numbers from a file
def read_numbers(filename):
    with open(filename, "r") as f:
        lines = f.readlines()
    
    # Convert text data to a 2D list of numbers
    data = [[int(num) for num in line.split()] for line in lines]
    return data

# Function to compute sum of each column
def column_sum(data):
    cols = len(data[0])
    sums = [0] * cols
    for row in data:
        for i in range(cols):
            sums[i] += row[i]
    return sums

# Function to compute mean of each column
def column_mean(data):
    sums = column_sum(data)
    count = len(data)
    return [s / count for s in sums]

# Function to compute median of each column
def column_median(data):
    cols = len(data[0])
    medians = []
    for i in range(cols):
        column_values = sorted(row[i] for row in data)
        n = len(column_values)
        if n % 2 == 1:
            medians.append(column_values[n // 2])
        else:
            medians.append((column_values[n // 2 - 1] + column_values[n // 2]) / 2)
    return medians

# Function to compute standard deviation of each column
def column_std_dev(data):
    means = column_mean(data)
    cols = len(data[0])
    std_devs = [0] * cols
    count = len(data)
    
    for row in data:
        for i in range(cols):
            std_devs[i] += (row[i] - means[i]) ** 2
    
    return [(s / count) ** 0.5 for s in std_devs]

# Function to compute frequency of each number in each column
def column_frequency(data):
    cols = len(data[0])
    freq_dicts = [{} for _ in range(cols)]
    
    for row in data:
        for i in range(cols):
            num = row[i]
            if num in freq_dicts[i]:
                freq_dicts[i][num] += 1
            else:
                freq_dicts[i][num] = 1
    return freq_dicts

# Read data
data = read_numbers("numbers.txt")

# Compute statistics
print("Sum of columns:", column_sum(data))
print("Mean of columns:", column_mean(data))
print("Median of columns:", column_median(data))
print("Standard Deviation of columns:", column_std_dev(data))

# Compute frequency
freq = column_frequency(data)
print("\nFrequency Count per column:")
for i, col_freq in enumerate(freq):
    print(f"Column {i+1}: {col_freq}")
